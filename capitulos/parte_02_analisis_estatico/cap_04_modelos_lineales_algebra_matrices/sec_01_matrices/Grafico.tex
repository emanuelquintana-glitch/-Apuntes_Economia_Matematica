\documentclass{standalone}
\usepackage[utf8]{inputenc}
\usepackage{tikz}
\usepackage{tikz-3dplot}
\usepackage{animate}

% Paleta de colores "High-Tech" 2026
\definecolor{fondo}{RGB}{5,10,15}
\definecolor{cianLuz}{RGB}{0,255,255}
\definecolor{oro}{RGB}{255,215,0}

\begin{document}
\pagecolor{fondo}

\begin{animateinline}[controls, loop, autoplay]{12}
\multiframe{30}{rAng=0+12}{
    % La cámara orbita lentamente cambiando el azimut (rAng)
    \tdplotsetmaincoords{70}{110 + \rAng*0.5}
    \begin{tikzpicture}[tdplot_main_coords, scale=2.5]

        % 1. EJES DE REFERENCIA
        \draw[white!20, ->] (0,0,0) -- (2.5,0,0) node[anchor=north east, font=\tiny] {$X_1$};
        \draw[white!20, ->] (0,0,0) -- (0,2.5,0) node[anchor=north west, font=\tiny] {$X_2$};
        \draw[white!20, ->] (0,0,0) -- (0,0,2.2) node[anchor=south, font=\tiny] {$Y$};

        % 2. SUPERFICIE TRIDIMENSIONAL (Silla de montar / Regresión no lineal)
        % Dibujamos la superficie mediante una malla de hilos para mayor claridad
        \foreach \x in {0,0.2,...,2}{
            \draw[cianLuz, opacity=0.4, ultra thin]
                plot[domain=0:2, samples=15] (\x, \x2, {0.5*(\x^2 - \x2^2) + 1});
        }
        \foreach \y in {0,0.2,...,2}{
            \draw[cianLuz, opacity=0.4, ultra thin]
                plot[domain=0:2, samples=15] (\x1, \y, {0.5*(\x1^2 - \y^2) + 1});
        }

        % 3. INDICADOR DINÁMICO (Autovector de convergencia)
        \pgfmathsetmacro{\posX}{1.2*cos(\rAng)}
        \pgfmathsetmacro{\posY}{1.2*sin(\rAng)}
        % Normalizamos para que se mantenga en el rango visible
        \fill[oro] (1,1,1) circle (1.5pt);
        \draw[oro, thick, dashed] (1,1,1) -- (1,1,0);

        % 4. HUD DE TELEMETRÍA (Datos en pantalla)
        \node[white, anchor=north west, font=\ttfamily\fontsize{4}{5}\selectfont] at (-1.5, 2, 2.5) {
            \begin{tabular}{l}
                \textcolor{cianLuz}{[3D\_SURFACE\_ANALYSIS]} \\
                AZIMUTH: \pgfmathparse{110 + \rAng*0.5}\pgfmathresult \\
                MODEL: NON\_LINEAR\_BETA \\
                STATUS: \ifnum\rAng>180 STABLE \else COMPUTING \fi
            \end{tabular}
        };

    \end{tikzpicture}
}
\end{animateinline}
\end{document}
